   227→/* ────────────────────────────────────────────────────────────────────────────
   228→ * Markdown rendering with manual styling (we don't ship @tailwindcss/typography).
   229→ * ──────────────────────────────────────────────────────────────────────────── */
   230→const markdownComponents: React.ComponentProps<typeof ReactMarkdown>["components"] = {
   231→  h1: ({ children }) => <h1 className="text-base font-bold mt-1 mb-1">{children}</h1>,
   232→  h2: ({ children }) => <h2 className="text-sm font-bold mt-3 mb-1">{children}</h2>,
   233→  h3: ({ children }) => <h3 className="text-sm font-semibold mt-2 mb-1">{children}</h3>,
   234→  p: ({ children }) => <p className="text-sm leading-relaxed">{children}</p>,
   235→  ul: ({ children }) => <ul className="list-disc pl-5 space-y-1 text-sm">{children}</ul>,
   236→  ol: ({ children }) => <ol className="list-decimal pl-5 space-y-1 text-sm">{children}</ol>,
   237→  li: ({ children }) => <li className="leading-relaxed">{children}</li>,
   238→  strong: ({ children }) => <strong className="font-semibold text-foreground">{children}</strong>,
   239→  em: ({ children }) => <em className="italic text-muted-foreground">{children}</em>,
   240→  code: ({ className, children, ...rest }) => {
   241→    // Fenced code blocks arrive with className="language-xxx"; render plain so
   242→    // the surrounding <pre> provides the styling. Inline code gets a chip.
   243→    const isBlock = /\blanguage-/.test(className || "");
   244→    if (isBlock) {
   245→      return (
   246→        <code className={cn("font-mono text-[12px]", className)} {...rest}>
   247→          {children}
   248→        </code>
   249→      );
   250→    }
   251→    return (
   252→      <code className="bg-muted text-foreground rounded px-1 py-0.5 text-[12px] font-mono">
   253→        {children}
   254→      </code>
   255→    );
   256→  },
   257→  pre: ({ children }) => (
   258→    <pre className="bg-muted/50 rounded-lg p-3 overflow-x-auto text-[12px] font-mono my-2">
   259→      {children}
   260→    </pre>
   261→  ),
   262→  a: ({ children, href }) => (
   263→    <a href={href} target="_blank" rel="noreferrer" className="text-primary underline underline-offset-2">
   264→      {children}
   265→    </a>
   266→  ),
   267→  blockquote: ({ children }) => (
   268→    <blockquote className="border-l-2 border-primary/40 pl-3 italic text-muted-foreground my-2">
   269→      {children}
   270→    </blockquote>
   271→  ),
   272→  hr: () => <hr className="border-border my-3" />,
   273→  table: ({ children }) => (
   274→    <div className="overflow-x-auto my-2">
   275→      <table className="w-full text-xs border-collapse">{children}</table>
   276→    </div>
   277→  ),
   278→  th: ({ children }) => <th className="border border-border px-2 py-1 text-left font-semibold">{children}</th>,
   279→  td: ({ children }) => <td className="border border-border px-2 py-1">{children}</td>,
   280→};
   281→

   283→  return (
   284→    <div className="flex items-center gap-2 px-3 py-2">
   285→      {[0, 1, 2].map((i) => (
   286→        <motion.span
   287→          key={i}
   288→          className="size-2 rounded-full bg-primary/70"
   289→          animate={{ y: [0, -4, 0], opacity: [0.4, 1, 0.4] }}
   290→          transition={{ duration: 0.9, repeat: Infinity, delay: i * 0.15, ease: "easeInOut" }}
   291→        />
   292→      ))}
   293→    </div>
   294→  );
   295→}

   297→/* ────────────────────────────────────────────────────────────────────────────
   298→ * AISection
   299→ * ──────────────────────────────────────────────────────────────────────────── */
   300→export function AISection() {
   301→  const [messages, setMessages] = React.useState<ChatMessage[]>([]);
   302→  const [input, setInput] = React.useState("");
   303→  const [selectedTool, setSelectedTool] = React.useState<string>("ideas");
   304→  const [loading, setLoading] = React.useState(false);
   305→  const [loadingHistory, setLoadingHistory] = React.useState(true);
   306→
   307→  const scrollAreaRef = React.useRef<HTMLDivElement>(null);
   308→  const inputRef = React.useRef<HTMLTextAreaElement>(null);
   309→
   310→  const tool = TOOLS[selectedTool] ?? TOOLS.ideas;
   311→
   312→  /* Load history on mount. */
   313→  React.useEffect(() => {
   314→    let cancelled = false;
   315→    (async () => {
   316→      try {
   317→        const data = await api<{ messages: ChatMessage[] }>("/api/ai/history");
   318→        if (!cancelled && Array.isArray(data.messages)) {
   319→          setMessages(data.messages);
   320→        }
   321→      } catch (e) {
   322→        console.error("[ai] history load failed", e);
   323→      } finally {
   324→        if (!cancelled) setLoadingHistory(false);
   325→      }
   326→    })();
   327→    return () => {
   328→      cancelled = true;
   329→    };
   330→  }, []);
   331→
   332→  /* Auto-scroll to bottom on new message / loading change. */
   333→  React.useEffect(() => {
   334→    const root = scrollAreaRef.current;
   335→    if (!root) return;
   336→    const viewport = root.querySelector<HTMLElement>(
   337→      "[data-slot='scroll-area-viewport']"
   338→    );
   339→    if (!viewport) return;
   340→    viewport.scrollTo({ top: viewport.scrollHeight, behavior: "smooth" });
   341→  }, [messages.length, loading]);
   342→
   343→  const focusInput = React.useCallback(() => {
   344→    requestAnimationFrame(() => inputRef.current?.focus());
   345→  }, []);
   346→
   347→  function selectTool(key: string) {
   348→    setSelectedTool(key);
   349→    focusInput();
   350→  }
   351→
   352→  async function send(content?: string) {
   353→    const text = (content ?? input).trim();
   354→    if (!text || loading) return;
   355→
   356→    const userMsg: ChatMessage = { role: "user", content: text, tool: selectedTool };
   357→    const history = messages
   358→      .filter((m) => m.content?.trim())
   359→      .slice(-12)
   360→      .map((m) => ({ role: m.role, content: m.content }));
   361→
   362→    setMessages((prev) => [...prev, userMsg]);
   363→    setInput("");
   364→    setLoading(true);
   365→
   366→    try {
   367→      const data = await api<{ content: string; saved: boolean }>("/api/ai", {
   368→        method: "POST",
   369→        body: JSON.stringify({ tool: selectedTool, input: text, history }),
   370→      });
   371→      const assistantMsg: ChatMessage = {
   372→        role: "assistant",
   373→        content: data.content,
   374→        tool: selectedTool,
   375→      };
   376→      setMessages((prev) => [...prev, assistantMsg]);
   377→    } catch (e) {
   378→      console.error("[ai] send failed", e);
   379→      toast.error("Failed to reach the AI assistant. Please try again.");
   380→      setMessages((prev) => [
   381→        ...prev,
   382→        {
   383→          role: "assistant",
   384→          content:
   385→            "I couldn't reach the AI service just now. Please check your connection and try again.",
   386→          tool: selectedTool,
   387→        },
   388→      ]);
   389→    } finally {
   390→      setLoading(false);
   391→      focusInput();
   392→    }
   393→  }
   394→
   395→  function onKeyDown(e: React.KeyboardEvent<HTMLTextAreaElement>) {
   396→    if (e.key === "Enter" && !e.shiftKey) {
   397→      e.preventDefault();
   398→      void send();
   399→    }
   400→  }
   401→
   402→  function clearHistory() {
   403→    if (messages.length === 0) return;
   404→    setMessages([]);
   405→    toast.success("Chat cleared (history is still saved in the database)");
   406→  }
   407→
   408→  return (
   409→    <div className="space-y-5">
   410→      <SectionHeader
   411→        title="AI Assistant"
   412→        description="Your AI co-pilot for every step of content creation"
   413→        icon="sparkles"
   414→        actions={
   415→          <Button variant="outline" size="sm" onClick={clearHistory} disabled={messages.length === 0}>
   416→            <Icon name="refresh" className="size-4" />
   417→            Clear chat
   418→          </Button>
   419→        }
   420→      />
   421→
   422→      {/* Tool grid */}
   423→      <motion.div
   424→        initial={{ opacity: 0, y: 10 }}
   425→        animate={{ opacity: 1, y: 0 }}
   426→        transition={{ duration: 0.35, ease: [0.2, 0.8, 0.2, 1] }}
   427→        className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 xl:grid-cols-7 gap-2"
   428→      >
   429→        {TOOL_ORDER.map((key, i) => {
   430→          const t = TOOLS[key];
   431→          const c = colorFor(t.color);
   432→          const active = selectedTool === key;
   433→          return (
   434→            <motion.button
   435→              key={key}
   436→              type="button"
   437→              onClick={() => selectTool(key)}
   438→              initial={{ opacity: 0, y: 8 }}
   439→              animate={{ opacity: 1, y: 0 }}
   440→              transition={{ duration: 0.3, delay: i * 0.025 }}
   441→              whileHover={{ y: -2 }}
   442→              whileTap={{ scale: 0.97 }}
   443→              className={cn(
   444→                "group relative flex flex-col items-start gap-2 rounded-xl border p-3 text-left transition-colors",
   445→                active
   446→                  ? "border-primary/60 bg-primary/5 shadow-sm"
   447→                  : "border-border/60 bg-card/50 hover:border-border hover:bg-accent/40"
   448→              )}
   449→              aria-pressed={active}
   450→              title={t.description}
   451→            >
   452→              <div
   453→                className={cn(
   454→                  "size-8 rounded-lg grid place-items-center shrink-0 transition-transform group-hover:scale-105",
   455→                  c.soft
   456→                )}
   457→              >
   458→                <Icon name={t.icon} className="size-4" />
   459→              </div>
   460→              <div className="min-w-0">
   461→                <p className="text-xs font-semibold truncate">{t.label}</p>
   462→                <p className="text-[11px] text-muted-foreground line-clamp-2 leading-snug mt-0.5">
   463→                  {t.description}
   464→                </p>
   465→              </div>
   466→              {active && (
   467→                <motion.span
   468→                  layoutId="tool-active-dot"
   469→                  className="absolute top-2 right-2 size-1.5 rounded-full bg-primary"
   470→                />
   471→              )}
   472→            </motion.button>
   473→          );
   474→        })}
   475→      </motion.div>
   476→
   477→      {/* Chat panel */}
   478→      <motion.div
   479→        initial={{ opacity: 0, y: 12 }}
   480→        animate={{ opacity: 1, y: 0 }}
   481→        transition={{ duration: 0.4, delay: 0.1, ease: [0.2, 0.8, 0.2, 1] }}
   482→      >
   483→        <Card className="glass overflow-hidden p-0 min-h-[55vh] h-[calc(100vh-22rem)] max-h-[82vh] flex flex-col border-border/60">
   484→          {/* Gradient header */}
   485→          <div className="relative shrink-0 px-4 sm:px-5 py-3 border-b border-border/60">
   486→            <div className="absolute inset-0 bg-gradient-to-r from-emerald-500/10 via-teal-500/5 to-amber-500/10 pointer-events-none" />
   487→            <div className="relative flex items-center justify-between gap-3">
   488→              <div className="flex items-center gap-2 min-w-0">
   489→                <div className="size-8 rounded-lg bg-primary/15 grid place-items-center shrink-0">
   490→                  <Icon name="sparkles" className="size-4 text-primary" />
   491→                </div>
   492→                <div className="min-w-0">
   493→                  <p className="text-sm font-semibold leading-tight truncate">AI Co-pilot</p>
   494→                  <p className="text-[11px] text-muted-foreground truncate">
   495→                    {messages.length === 0
   496→                      ? "Ready when you are"
   497→                      : `${messages.length} message${messages.length === 1 ? "" : "s"}`}
   498→                  </p>
   499→                </div>
   500→              </div>
   501→              <div className="flex items-center gap-2 shrink-0">
   502→                {loading && (
   503→                  <span className="inline-flex items-center gap-1.5 text-[11px] text-muted-foreground">
   504→                    <span className="size-1.5 rounded-full bg-emerald-500 animate-pulse" />
   505→                    thinking
   506→                  </span>
   507→                )}
   508→              </div>
   509→            </div>
   510→          </div>
   511→
   512→          {/* Messages area */}
   513→          <div className="flex-1 min-h-0 relative">
   514→            <ScrollArea ref={scrollAreaRef} className="h-full">
   515→              <div className="px-4 sm:px-5 py-4 space-y-4">
   516→                <AnimatePresence initial={false}>
   517→                  {loadingHistory && messages.length === 0 ? (
   518→                    <div className="flex items-center justify-center py-16">
   519→                      <div className="flex items-center gap-2 text-sm text-muted-foreground">
   520→                        <Icon name="refresh" className="size-4 animate-spin" />
   521→                        Loading chat history…
   522→                      </div>
   523→                    </div>
   524→                  ) : messages.length === 0 && !loading ? (
   525→                    <EmptyHero
   526→                      toolKey={selectedTool}
   527→                      onPickExample={(ex) => {
   528→                        setInput(ex);
   529→                        focusInput();
   530→                      }}
   531→                    />
   532→                  ) : null}
   533→
   534→                  {messages.map((m, i) => (
   535→                    <MessageBubble key={m.id ?? `${i}-${m.role}`} message={m} />
   536→                  ))}
   537→
   538→                  {loading && (
   539→                    <motion.div
   540→                      initial={{ opacity: 0, y: 6 }}
   541→                      animate={{ opacity: 1, y: 0 }}
   542→                      exit={{ opacity: 0 }}
   543→                      className="flex items-start gap-3"
   544→                    >
   545→                      <Avatar tool={selectedTool} />
   546→                      <div className="glass rounded-2xl rounded-tl-sm border border-border/60">
   547→                        <TypingDots />
   548→                      </div>
   549→                    </motion.div>
   550→                  )}
   551→                </AnimatePresence>
   552→              </div>
   553→            </ScrollArea>
   554→          </div>
   555→
   556→          {/* Input area */}
   557→          <div className="shrink-0 border-t border-border/60 bg-card/40 px-3 sm:px-4 py-3">
   558→            <div className="flex items-center gap-2 mb-2 flex-wrap">
   559→              <Pill color={tool.color} icon={tool.icon}>
   560→                {tool.label}
   561→              </Pill>
   562→              <button
   563→                type="button"
   564→                onClick={() => setSelectedTool("ideas")}
   565→                className="inline-flex items-center justify-center size-4 rounded-full text-muted-foreground hover:text-foreground hover:bg-accent transition-colors"
   566→                aria-label="Deselect tool"
   567→              >

   672→function MessageBubble({ message }: { message: ChatMessage }) {
   673→  const isUser = message.role === "user";
   674→  const t = message.tool ? TOOLS[message.tool] : undefined;
   675→  const c = t ? colorFor(t.color) : colorFor("emerald");
   676→

   679→      initial={{ opacity: 0, y: 8 }}
   680→      animate={{ opacity: 1, y: 0 }}
   681→      transition={{ duration: 0.25, ease: [0.2, 0.8, 0.2, 1] }}
   682→      className={cn("flex items-start gap-3", isUser ? "flex-row-reverse" : "flex-row")}
   683→    >
   684→      {/* Avatar */}
   685→      <div
   686→        className={cn(
   687→          "size-8 rounded-lg grid place-items-center shrink-0",
   688→          isUser ? "bg-primary/15 text-primary" : c.soft
   689→        )}
   690→      >
   691→        <Icon name={isUser ? "users2" : t?.icon ?? "sparkles"} className="size-4" />
   692→      </div>
   693→
   694→      {/* Bubble */}
   695→      <div
   696→        className={cn(
   697→          "max-w-[85%] sm:max-w-[78%] rounded-2xl px-3.5 py-2.5",
   698→          isUser
   699→            ? "bg-primary/15 text-foreground rounded-tr-sm"
   700→            : "glass border border-border/60 rounded-tl-sm"
   701→        )}
   702→      >
   703→        {isUser ? (
   704→          <p className="text-sm leading-relaxed whitespace-pre-wrap break-words">{message.content}</p>
   705→        ) : (
   706→          <div className="text-sm leading-relaxed space-y-3">
   707→            <ReactMarkdown components={markdownComponents}>{message.content}</ReactMarkdown>
   708→          </div>
   709→        )}
   710→        {t && !isUser && (
   711→          <div className="mt-2 pt-2 border-t border-border/40 flex items-center gap-1.5">
   712→            <span className={cn("inline-flex items-center gap-1 text-[10px] font-medium", c.text)}>
   713→              <Icon name={t.icon} className="size-3" />
   714→              {t.label}
   715→            </span>
   716→          </div>
   717→        )}
   718→      </div>
   719→    </motion.div>
   720→  );
   721→}
